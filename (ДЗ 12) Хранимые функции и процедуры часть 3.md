**Функция** 

    CREATE OR REPLACE FUNCTION update_good_sum_mart()
    RETURNS TRIGGER AS $$
    DECLARE
        v_old_price NUMERIC(12,2) := NULL;
        v_new_price NUMERIC(12,2) := NULL;
        v_good_name VARCHAR(63) := NULL;
    BEGIN
        IF TG_OP = 'INSERT' THEN
        SELECT good_name, good_price 
        INTO v_good_name, v_new_price
        FROM goods WHERE goods_id = NEW.good_id;

        INSERT INTO good_sum_mart (good_name, sum_sale)
        VALUES (v_good_name, NEW.sales_qty * v_new_price)
        ON CONFLICT (good_name) DO UPDATE
        SET sum_sale = good_sum_mart.sum_sale + NEW.sales_qty * v_new_price;

    ELSIF TG_OP = 'UPDATE' THEN
        SELECT good_price, good_name
        INTO v_old_price, v_good_name
        FROM goods WHERE goods_id = OLD.good_id;

        SELECT good_price
        INTO v_new_price
        FROM goods WHERE goods_id = NEW.good_id;

        UPDATE good_sum_mart
        SET sum_sale = sum_sale
            - OLD.sales_qty * v_old_price
            + NEW.sales_qty * v_new_price
        WHERE good_name = v_good_name;

    ELSIF TG_OP = 'DELETE' THEN
        SELECT good_price, good_name
        INTO v_old_price, v_good_name
        FROM goods WHERE goods_id = OLD.good_id;

        UPDATE good_sum_mart
        SET sum_sale = sum_sale - OLD.sales_qty * v_old_price
        WHERE good_name = v_good_name;
    END IF;

    RETURN NULL;
    END;
    $$ LANGUAGE plpgsql;

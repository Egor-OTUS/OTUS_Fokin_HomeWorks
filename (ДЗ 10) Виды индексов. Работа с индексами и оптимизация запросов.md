***Написать комментарии к каждому из индексов***  
Комментарии к индексам создавались по мере создания самих индексов и отображены на скриншотах

Структура

   CREATE TABLE employees (  
        id SERIAL PRIMARY KEY,  
        first_name VARCHAR(50),  
        last_name VARCHAR(50),  
        department VARCHAR(50),  
       salary NUMERIC(10, 2),  
       hire_date DATE,  
       bio TEXT  
   );

   INSERT INTO employees (first_name, last_name, department, salary, hire_date, bio)  
   SELECT  
       'Name' || g,  
       'Surname' || g,  
       CASE WHEN g % 3 = 0 THEN 'IT'  
            WHEN g % 3 = 1 THEN 'HR'  
            ELSE 'Finance' END,  
       50000 + (random() * 50000),  
       CURRENT_DATE - (random() * 3650)::int,  
       'Biography of employee ' || g || '. Lorem ipsum dolor sit amet...'  
   FROM generate_series(1, 10000) g;  

   CREATE INDEX idx_employees_last_name ON employees (last_name);
